## questions.js 再生成コマンド（全回解説完了後）
```python
import json, glob

all_questions = []
for filepath in sorted(glob.glob('/Users/segawashun/Documents/claude/あんまアプリ/quiz-app/data/questions_*.json')):
    with open(filepath, 'r', encoding='utf-8') as f:
        all_questions.extend(json.load(f))

with open('/Users/segawashun/Documents/claude/あんまアプリ/quiz-app/data/questions.js', 'w', encoding='utf-8') as f:
    f.write('// 問題データ（全回分）\n')
    f.write('const QUESTIONS_DATA = ')
    json.dump(all_questions, f, ensure_ascii=False, indent=2)
    f.write(';\n')

print(f"questions.js updated: {len(all_questions)} questions")
```
